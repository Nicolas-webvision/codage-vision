Le choix des barres de recherche

Les barres de recherche sont des éléments essentiels pour améliorer l'expérience utilisateur sur un site web. Elles permettent aux visiteurs de trouver rapidement des informations spécifiques en effectuant une recherche ciblée, ce qui peut considérablement augmenter l'efficacité et la convivialité du site. Un design et une fonctionnalité optimisés pour les barres de recherche peuvent transformer une simple visite en une interaction réussie, tout en aidant à réduire la frustration des utilisateurs.

document.addEventListener('DOMContentLoaded', function() {
const searchInput = document.getElementById('search-input');
const searchResults = document.getElementById('search-results');

// Simulated data for demonstration
const data = [
    'Planète Mars',
    'Planète Vénus',
    'Planète Terre',
    'Planète Jupiter',
    'Planète Saturne',
    'Planète Uranus',
    'Planète Neptune'
];

function updateResults(query) {
    searchResults.innerHTML = '';
    if (query) {
        const filteredData = data.filter(item =>
            item.toLowerCase().includes(query.toLowerCase())
        );

        if (filteredData.length > 0) {
            searchResults.style.display = 'block';
            filteredData.forEach(item => {
                const li = document.createElement('li');
                li.textContent = item;
                li.addEventListener('click', function() {
                    searchInput.value = item;
                    searchResults.style.display = 'none';
                });
                searchResults.appendChild(li);
            });
        } else {
            searchResults.style.display = 'none';
        }
    } else {
        searchResults.style.display = 'none';
    }
}

searchInput.addEventListener('input', function() {
    updateResults(searchInput.value);
});

document.addEventListener('click', function(event) {
    if (!searchInput.contains(event.target) && !searchResults.contains(event.target)) {
        searchResults.style.display = 'none';
    }
});
});
